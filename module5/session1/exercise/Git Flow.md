# A. From Scratch

1. git init
2. git add /path/to/file
3. git commit -s
4. git remote add 'remote-name' 'link'
5. git push 'remote-name' master

# B. From Existing Project

1. git clone 'link'
2. git checkout -b 'branch-name'
3. git add /path/to/file
4. git commit -s
5. git push 'remote-name' 'branch-name'
